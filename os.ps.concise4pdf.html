1  // Share section
2  int number[n] = {0};
1  // Each process
2  number[i] = max(number, n)+1;
3  for(j = (i+1) % n; j != i; j = (j+1) % n)
4    while(number[i] > number[j] && number[j] != 0)
5        ;
6  /* Critical Section */
7  number[i] = 0;
